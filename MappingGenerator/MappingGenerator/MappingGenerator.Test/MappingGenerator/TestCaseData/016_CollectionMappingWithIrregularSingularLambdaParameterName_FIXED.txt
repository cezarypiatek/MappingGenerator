using System;
using System.Collections.Generic;
using System.Linq;

namespace TestAutoMapper.Identity.X
{
    public class XX
    {
        public int Id { get; set; }
        public string Name { get; set; }
    }

    public class YY
    {
        public int Id { get; set; }
        public string Name { get; set; }
    }

    public static class Mapper{

        public static List<YY> Map(List<XX> people)
        {
            return people.ConvertAll(person => new YY
            {
                Id = person.Id,
                Name = person.Name
            });
        }
    }
}
