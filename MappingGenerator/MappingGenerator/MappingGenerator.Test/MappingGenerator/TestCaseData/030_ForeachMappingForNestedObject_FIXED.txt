using System;
using System.Collections.Generic;
using System.Text;
using System.Collections.ObjectModel;

namespace MappingGenerator.Test.MappingGenerator.TestCaseData
{
    public class TestMapper
    {
        public static UserDTO Map(UserEntity entity)
        {
            var userDTO = new UserDTO()
            {
                Age = entity.Age
            };
            foreach (var account in entity.Accounts)
            {
                var accountDTO = new AccountDTO()
                {
                    BankName = account.BankName,
                    Number = account.Number
                };
                foreach (var address in account.Addresses)
                {
                    var addressDTO = new AddressDTO()
                    {
                        City = address.City,
                        ZipCode = address.ZipCode
                    };
                    accountDTO.Addresses.Add(addressDTO);
                }

                userDTO.Accounts.Add(accountDTO);
            }

            return userDTO;
        }
    }

    public class UserDTO
    {
        public int Age { get; set; }
        public List<AccountDTO> Accounts { get; } = new List<AccountDTO>();
    }

    public class UserEntity
    {
        public int Age { get; set; }
        public List<AccountEntity> Accounts { get; set; }
    }

    public class AccountDTO
    {
        public string BankName { get; set; }
        public string Number { get; set; }
        public List<AddressDTO> Addresses { get; } = new List<AddressDTO>();
    }

    public class AccountEntity
    {
        public string BankName { get; set; }
        public string Number { get; set; }
        public List<AddressEntity> Addresses { get; set; }
    }

    public class AddressDTO
    {
        public string City { get; set; }
        public string ZipCode { get; set; }
    }

    public class AddressEntity
    {
        public string City { get; set; }
        public string ZipCode { get; set; }
    }
}
